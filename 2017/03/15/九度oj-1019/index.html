九度oj-1019

发表于 2017-03-15   |   分类于 九度oj   |  

题目描述：

读入一个只包含 +, -, *, / 的非负整数计算表达式，计算该表达式的值。

输入：

测试输入包含若干测试用例，每个测试用例占一行，每行不超过200个字符，整数和运算符之间用一个空格分隔。没有非法表达式。当一行中只有0时输入结束，相应的结果不要输出。

输出：

对每个测试用例输出1行，即该表达式的值，精确到小数点后2位。

样例输入：

1 + 2
4 + 2 * 5 - 7 / 11
0

样例输出：

3.00
13.36

代码：

# include <iostream>  
# include <string.h>  
# include <stdio.h>  
# include <stack>  
# include <stdlib.h>  
using namespace std;  
int main()  
{  
char test[201],num[200],sym[4]={'*','/','+','-'};  
int len;  
stack<double> cal;  
while(cin.getline(test,201))  
{  
    len = strlen(test);  
    if(len==1&&test[0]=='0') break;  
    int flag=2,count=-1;  
    for(int i=0;i<len;i++)  
    {  
        int flag2=0;//帮助判断是否是数字并存入num   
        if(test[i]==' ') continue;//没注意题目中整数和运算符用空格分隔，去掉空格。   
        if(test[i]>='0'&&test[i]<='9')  
        {  
            count++;  
            num[count]=test[i];  
            flag2=1;  
        }  
        if(flag2==0||i==len-1)  
        {  
            num[count+1]='\0';  
            double n = atoi(num);  
            count=-1;  
            if(flag==0) //乘法 相乘后放入栈   
            {  
                double m=cal.top();  
                n*=m;  
                cal.pop();  
                cal.push(n);  
            }  
            else if(flag==1) //除法  相除后放入栈   
            {  
                double m=cal.top();  
                n = m/n;  
                cal.pop();  
                cal.push(n);  
            }  
            else if(flag==3) //减法 存为负数放入栈   
            {  
                cal.push(-n);  
            }  
            else if(flag==2) //加法  直接放入栈   
            {  
                cal.push(n);  
            }   
            for(int j =0;j<4;j++) //判断算数符号   
            {  
                if(test[i]==sym[j]) flag=j;  
            }       
        }  

    }  
    double alcount=0;  
    while(cal.empty()!=1)//栈内剩余数相加   
    {  
        alcount+=cal.top();  
        cal.pop();  
    }      
    printf("%.2lf\n",alcount);    

}  
return 0;  
}

作者提醒

本题思路：

先计算乘除后计算加减。将数字依次存入栈中，遇到乘号则先相乘再存入栈中，
遇到除号则相除再入栈，遇到减号则将数字存成负数入栈。

最后将栈内元素相加。

由于做的时候没注意数字和运算符间有空格，导致WA，于是强行加了一行判断空格。。。
